model User {
  id                    String               @id @default(uuid())
  email                 String               @unique
  passwordHash          String
  avatar                String?
  userType              UserType             @default(individual)
  role                  UserRole             @default(VIEWER)
  tier                  UserTier             @default(Free)
  quota                 Json?
  defaultOrganizationId String?
  createdAt             DateTime             @default(now())
  updatedAt             DateTime             @updatedAt
  apiKeys               ApiKey[]             @relation("UserApiKeys")
  auditLogs             AuditLog[]           @relation("UserAuditLogs")
  billingEvents         BillingEvent[]       @relation("BillingEventUser")
  memberships           Membership[]         @relation("UserMemberships")
  organizationMembers   OrganizationMember[] @relation("UserOrganizationMember")
  ownedOrganizations    Organization[]       @relation("OrganizationOwner")
  ownedProjects         Project[]            @relation("ProjectOwner")
  reviewLogs            PublishingReview[]   @relation("ReviewerUser")
  subscriptions         Subscription[]       @relation("UserSubscription")

  @@map("users")
}

model Organization {
  id            String               @id @default(uuid())
  name          String
  ownerId       String
  slug          String?              @unique
  createdAt     DateTime             @default(now())

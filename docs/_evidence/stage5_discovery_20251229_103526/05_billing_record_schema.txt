model BillingRecord {
  id             String   @id @default(uuid())
  userId         String
  organizationId String?
  planId         String?
  amount         Float
  credits        Float
  computeSeconds Float
  periodStart    DateTime
  periodEnd      DateTime
  status         String
  invoiceId      String?
  createdAt      DateTime @default(now())
  updatedAt      DateTime @updatedAt

  @@map("billing_records")
}

model Model {
  id        String   @id @default(uuid())
  name      String
  type      String
  version   String
  provider  String?
  config    Json?
  isActive  Boolean  @default(true)
  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt

  @@unique([name, version])
  @@map("models")

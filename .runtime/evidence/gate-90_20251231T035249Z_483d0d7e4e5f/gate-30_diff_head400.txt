+import { Injectable, Logger } from '@nestjs/common';
+import { PrismaService } from '../prisma/prisma.service';
+
+/**
+ * 价格配置 (Credits)
+ * 1 Credit = 0.05 RMB
+ */
+export const PRICING_CONFIG = {
+  // GPU 推理 (Credits per second)
+  // 参考: 1 小时 = 8 Credits (即 0.4 元/小时)
+  GPU_PER_SEC: 8 / 3600,
+
+  // LLM Token (Credits per 1k tokens)
+  // 参考: 1k tokens = 0.02 元 = 0.4 Credits
+  TOKEN_PER_1K: 0.4,
+
+  // 存储 (Credits per GB per hour)
+  STORAGE_PER_GB_HR: 0.1,
+};
+
+@Injectable()
+export class CeCostService {
+  private readonly logger = new Logger(CeCostService.name);
+
+  constructor(private readonly prisma: PrismaService) { }
+
+  /**
+   * 根据资源消耗计算成本 (Credits)
+   */
+  calculateCost(type: string, quantity: number, modelFactor = 1.0): number {
+    let baseCost = 0;
+
+    switch (type.toUpperCase()) {
+      case 'GPU':
+        baseCost = quantity * PRICING_CONFIG.GPU_PER_SEC;
+        break;
+      case 'TOKEN':
+        baseCost = (quantity / 1000) * PRICING_CONFIG.TOKEN_PER_1K;
+        break;
+      case 'STORAGE':
+        baseCost = quantity * PRICING_CONFIG.STORAGE_PER_GB_HR;
+        break;
+      default:
+        return 0;
+    }
+
+    return baseCost * modelFactor;
+  }
+
+  /**
+   * 原子化写入账本 (幂等去重)
+   */
+  async recordLedger(data: {
+    traceId: string;
+    projectId: string;
+    orgId: string;
+    userId?: string;
+    costType: string;
+    quantity: number;
+    unitCost: number;
+    totalCost: number;
+    timestamp?: Date;
+  }) {
+    try {
+      // 检查是否已计费 (同一个 traceId 下同一种类型的成本只允许计费一次)
+      const existing = await this.prisma.costLedger.findFirst({
+        where: {
+          traceId: data.traceId,
+          costType: data.costType,
+        },
+      });
+
+      if (existing) {
+        this.logger.debug(
+          `[Cost] Trace ${data.traceId} type ${data.costType} already charged, skipping.`
+        );
+        return existing;
+      }
+
+      return await this.prisma.costLedger.create({
+        data: {
+          traceId: data.traceId,
+          projectId: data.projectId,
+          orgId: data.orgId,
+          userId: data.userId || null,
+          costType: data.costType,
+          unitCost: data.unitCost,
+          quantity: data.quantity,
+          totalCost: data.totalCost,
+          timestamp: data.timestamp || new Date(),
+        },
+      });
+    } catch (error: unknown) {
+      const err = error as Error;
+      this.logger.error(`Failed to record cost ledger for trace ${data.traceId}`, err.stack);
+      throw error;
+    }
+  }
+}

+import { Injectable, Logger } from '@nestjs/common';
+import { Cron, CronExpression } from '@nestjs/schedule';
+import { PrismaService } from '../prisma/prisma.service';
+import { CeCostService, PRICING_CONFIG } from './ce-cost.service';
+
+@Injectable()
+export class CostIngestJob {
+  private readonly logger = new Logger(CostIngestJob.name);
+  private isProcessing = false;
+
+  constructor(
+    private readonly prisma: PrismaService,
+    private readonly ceCostService: CeCostService
+  ) { }
+
+  /**
+   * 每 5 分钟扫描一次最近的遥感 COST 记录并处理入账
+   * Stage-8 P1: 异步旁路入账
+   */
+  @Cron(CronExpression.EVERY_5_MINUTES)
+  async handleCron() {
+    if (this.isProcessing) return;
+    this.isProcessing = true;
+
+    try {
+      this.logger.log('[CostIngest] Starting cost ingestion task...');
+
+      // 1. 获取最近 10 分钟未处理或需要重新审计的 COST 记录
+      // 注意：由于 P1 采用幂等 upsert 逻辑，我们可以稍微重叠扫描范围增加可靠性
+      const records = await this.prisma.telemetryRecord.findMany({
+        where: {
+          kind: 'COST',
+          timestamp: {
+            gte: new Date(Date.now() - 10 * 60 * 1000), // 最近 10 分钟
+          },
+        },
+      });
+
+      this.logger.log(`[CostIngest] Found ${records.length} COST telemetry records to process.`);
+
+      let processedCount = 0;
+      for (const record of records) {
+        // 2. 构造归因上下文 (从 payload 提取)
+        const payload = (record.payload as Record<string, unknown>) || {};
+        const projectId = payload.projectId as string | undefined;
+        const orgId = (payload.orgId || payload.organizationId) as string | undefined;
+
+        if (!projectId || !record.traceId) {
+          this.logger.warn(
+            `[CostIngest] Record ${record.id} missing projectId or traceId, skipping.`
+          );
+          continue;
+        }
+
+        // 3. 计算并写入 (CeCostService 内部处理幂等)
+        const quantity = record.value || 0;
+        const unitCost = this.getUnitCost(record.name);
+        const modelFactor = (payload.modelFactor as number | undefined) || 1.0;
+        const totalCost = this.ceCostService.calculateCost(record.name, quantity, modelFactor);
+
+        if (totalCost > 0) {
+          await this.ceCostService.recordLedger({
+            traceId: record.traceId,
+            projectId,
+            orgId: orgId || 'system',
+            userId: payload.userId as string | undefined,
+            costType: record.name,
+            quantity,
+            unitCost,
+            totalCost,
+            timestamp: record.timestamp,
+          });
+          processedCount++;
+        }
+      }
+
+      this.logger.log(`[CostIngest] Successfully processed ${processedCount} cost entries.`);
+    } catch (error: unknown) {
+      const err = error as Error;
+      this.logger.error('[CostIngest] Failed during cost ingestion', err.stack);
+    } finally {
+      this.isProcessing = false;
+    }
+  }
+
+  private getUnitCost(type: string): number {
+    switch (type.toUpperCase()) {
+      case 'GPU':
+        return PRICING_CONFIG.GPU_PER_SEC;
+      case 'TOKEN':
+        return PRICING_CONFIG.TOKEN_PER_1K;
+      case 'STORAGE':
+        return PRICING_CONFIG.STORAGE_PER_GB_HR;
+      default:
+        return 0;
+    }
+  }
+}

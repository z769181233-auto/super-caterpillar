+#!/usr/bin/env bash
+set -euo pipefail
+
+ROOT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")/../.." && pwd)"
+cd "$ROOT_DIR"
+
+TS="$(date +"%Y%m%d_%H%M%S")"
+EVID_ROOT="docs/_evidence/full_audit_${TS}"
+mkdir -p "$EVID_ROOT"
+
+# --- Tunables ---
+T_FAST="${T_FAST:-120}"     # quick commands
+T_MED="${T_MED:-300}"       # medium commands
+T_LONG="${T_LONG:-900}"     # long commands (build/dev/gates)
+API_PORT="${API_PORT:-3001}"
+API_BASE="${API_BASE:-http://127.0.0.1:${API_PORT}}"
+
+# --- Helpers ---
+log(){ echo "[$(date +"%H:%M:%S")] $*" | tee -a "$EVID_ROOT/00_run.log"; }
+
+dump_diag() {
+  local name="$1"
+  local dir="$EVID_ROOT/hang_diag_${name}_$(date +"%H%M%S")"
+  mkdir -p "$dir"
+  {
+    echo "=== DIAG $(date -Iseconds) ==="
+    echo "PWD=$(pwd)"
+    echo "--- git ---"
+    git status -sb || true
+    echo "--- ports ---"
+    lsof -nP -iTCP:${API_PORT} -sTCP:LISTEN || true
+    echo "--- process (node/pnpm) ---"
+    ps aux | rg -n "node|pnpm|next|nest|ts-node|prisma" || true
+    echo "--- process tree (best-effort) ---"
+    pgrep -fl "pnpm|next|nest|node|ts-node" || true
+    echo "--- docker (if any) ---"
+    docker ps >/dev/null 2>&1 && docker ps || true
+  } > "$dir/diag.txt" 2>&1
+
+  # Tail common logs if exist
+  for f in "$EVID_ROOT"/*.log "$EVID_ROOT"/*/*.log; do
+    [[ -f "$f" ]] || continue
+    tail -n 200 "$f" > "$dir/$(basename "$f").tail200.txt" 2>/dev/null || true
+  done
+
+  echo "$dir" > "$EVID_ROOT/last_hang_diag_dir.txt"
+}
+
+kill_tree_by_port() {
+  local port="$1"
+  local pids
+  pids="$(lsof -t -iTCP:${port} -sTCP:LISTEN 2>/dev/null || true)"
+  [[ -n "$pids" ]] || return 0
+  log "[KILL] killing listeners on port ${port}: ${pids}"
+  kill -TERM $pids 2>/dev/null || true
+  sleep 1
+  kill -KILL $pids 2>/dev/null || true
+}
+
+kill_by_pattern() {
+  local pat="$1"
+  local pids
+  pids="$(pgrep -f "$pat" || true)"
+  [[ -n "$pids" ]] || return 0
+  log "[KILL] killing by pattern '${pat}': ${pids}"
+  kill -TERM $pids 2>/dev/null || true
+  sleep 1
+  kill -KILL $pids 2>/dev/null || true
+}
+
+run_step() {
+  local name="$1"
+  local timeout_s="$2"
+  shift 2
+  log "=== STEP ${name} (timeout=${timeout_s}s) ==="
+  local out="$EVID_ROOT/${name}.log"
+  set +e
+  # Prefer repo's timeout runner if exists
+  if [[ -x "tools/dev/run_with_timeout.sh" ]]; then
+    bash tools/dev/run_with_timeout.sh "$timeout_s" "$@" >"$out" 2>&1
+  else
+    # fallback (macOS)
+    perl -e 'alarm shift; exec @ARGV' "$timeout_s" "$@" >"$out" 2>&1
+  fi
+  local rc=$?
+  set -e
+  if [[ $rc -ne 0 ]]; then
+    log "[FAIL] STEP ${name} rc=$rc"
+    dump_diag "$name"
+    # hard cleanup
+    kill_tree_by_port "$API_PORT"
+    kill_by_pattern "pnpm --filter api run dev"
+    kill_by_pattern "next dev|next start|nest start|nest build|ts-node|prisma"
+    return $rc
+  fi
+  log "[PASS] STEP ${name}"
+}
+
+# --- Begin ---
+log "EVID_ROOT=$EVID_ROOT"
+log "API_PORT=$API_PORT API_BASE=$API_BASE"
+
+# 0) preflight
+run_step "01_git_status" "$T_FAST" git status -sb
+run_step "02_versions" "$T_FAST" bash -c 'node -v && pnpm -v && git --version'
+
+# 1) static checks
+run_step "10_typecheck" "$T_MED" pnpm -w run typecheck
+run_step "11_build" "$T_LONG" pnpm -w run build
+
+# 2) prisma/db checks (best-effort; do not assume migrate)
+run_step "20_prisma_generate" "$T_MED" pnpm --filter database run build
+run_step "21_prisma_validate" "$T_MED" bash -c 'pnpm --filter api run prisma:validate 2>/dev/null || true'
+
+# 3) run key gates (add/remove based on your repo)
+GATES=(
+  "tools/gate/gates/gate-stage7_ce07_memory.sh"
+  "tools/gate/gates/gate-d5_pipeline.sh"
+  "tools/gate/gates/gate-ce_video_e2e.sh"
+)
+for g in "${GATES[@]}"; do
+  if [[ -f "$g" ]]; then
+    run_step "30_gate_$(basename "$g")" "$T_LONG" bash "$g"
+  else
+    log "[SKIP] missing gate: $g"
+  fi
+done
+
+# 4) final cleanup check
+kill_tree_by_port "$API_PORT" || true

+import { Injectable, Logger, Inject } from '@nestjs/common';
+import { PrismaService } from '../prisma/prisma.service';
+import { BudgetWindowService } from '../billing/budget-window.service';
+import { OwnerType, BudgetAction, BudgetWindow } from 'database';
+
+@Injectable()
+export class PlanService {
+    private readonly logger = new Logger(PlanService.name);
+
+    constructor(
+        @Inject(PrismaService) private readonly prisma: PrismaService,
+        @Inject(BudgetWindowService) private readonly budgetWindowService: BudgetWindowService,
+    ) { }
+
+    /**
+     * 将 Plan 绑定到 Project
+     * 核心逻辑：Plan -> Quotas -> Budgets (Runtime Core)
+     * 实现 "产品层变动自动同步到内核隔离层"
+     */
+    async assignPlanToProject(projectId: string, planCode: string) {
+        const p = await this.prisma.plan.findUnique({
+            where: { code: planCode },
+            include: { quotas: true },
+        });
+
+        if (!p) {
+            throw new Error(`Plan ${planCode} not found`);
+        }
+
+        // 1. 记录 ProjectPlan 绑定关系
+        await this.prisma.$transaction(async (tx) => {
+            // Upsert ProjectPlan
+            await tx.projectPlan.upsert({
+                where: { projectId },
+                create: {
+                    projectId,
+                    planId: p.id,
+                },
+                update: {
+                    planId: p.id,
+                    startedAt: new Date(), // Reset start time on change
+                },
+            });
+
+            // 2.【关键】将 PlanQuotas 转换为 Runtime Budgets
+            // 这是连接 Product Layer 和 Kernel Layer 的唯一桥梁
+            for (const quota of p.quotas) {
+                let window: BudgetWindow | null = null;
+
+                if (quota.limitType === 'DAILY_CREDITS') window = BudgetWindow.DAILY;
+                if (quota.limitType === 'MONTHLY_CREDITS') window = BudgetWindow.MONTHLY;
+                // CONCURRENCY 等其他限制暂不映射为 Financial Budget
+
+                if (window) {
+                    await tx.budget.upsert({
+                        where: {
+                            targetId_window: {
+                                targetId: projectId,
+                                window: window,
+                            }
+                        },
+                        create: {
+                            targetId: projectId,
+                            targetType: OwnerType.PROJECT,
+                            window: window,
+                            limitCredits: quota.limitValue,
+                            action: quota.action,
+                        },
+                        update: {
+                            limitCredits: quota.limitValue,
+                            action: quota.action,
+                            // 注意：这里更新了限额，BudgetWindowService 在下一次 check 时会自动使用新限额
+                        },
+                    });
+                    this.logger.log(`[PlanService] Synced budget for project ${projectId}: ${window} = ${quota.limitValue}`);
+                }
+            }
+        });
+
+        this.logger.log(`[PlanService] Assigned plan ${planCode} to project ${projectId}`);
+    }
+
+    /**
+     * 创建基础套餐 (Seed)
+     */
+    async ensureSeedPlans() {
+        const free = await this.prisma.plan.upsert({
+            where: { code: 'FREE' },
+            create: {
+                code: 'FREE',
+                name: 'Free Tier',
+                price: 0,
+            },
+            update: {}
+        });
+
+        // Free: Daily 100 Credits
+        await this.prisma.planQuota.createMany({
+            data: [
+                { planId: free.id, limitType: 'DAILY_CREDITS', limitValue: 100, action: BudgetAction.BLOCK }
+            ],
+            skipDuplicates: true
+        });
+
+        const pro = await this.prisma.plan.upsert({
+            where: { code: 'PRO' },
+            create: {
+                code: 'PRO',
+                name: 'Pro Tier',
+                price: 29.99,
+            },
+            update: {}
+        });
+
+        // Pro: Daily 1000 Credits
+        await this.prisma.planQuota.createMany({
+            data: [
+                { planId: pro.id, limitType: 'DAILY_CREDITS', limitValue: 1000, action: BudgetAction.BLOCK }
+            ],
+            skipDuplicates: true
+        });
+    }

+import { Injectable, Logger, Inject } from '@nestjs/common';
+import { PrismaService } from '../prisma/prisma.service';
+import { CreditLedgerService } from '../billing/credit-ledger.service';
+import { CreditAction } from 'database';
+
+@Injectable()
+export class StatementService {
+    private readonly logger = new Logger(StatementService.name);
+
+    constructor(
+        @Inject(PrismaService) private readonly prisma: PrismaService,
+    ) { }
+
+    /**
+     * 生成账单 (Draft)
+     * SSOT: CostLedger (SETTLE action amounts)
+     * 
+     * 逻辑：
+     * 1. 确定周期 (Start, End)
+     * 2. 聚合该周期内所有 SETTLE 类型的 Ledger 记录 (actualCost)
+     * 3. 创建 BillingStatement
+     */
+    async generateStatement(orgId: string, periodStart: Date, periodEnd: Date) {
+        // 1. 聚合 Cost
+        // 使用 Prisma 的 aggregate
+        const aggregation = await this.prisma.creditLedger.aggregate({
+            _sum: {
+                actualCost: true, // 核心：只统计实际结算成本，不统计 Reserve/Release
+            },
+            where: {
+                walletId: {
+                    in: await this.getOrgWalletIds(orgId),
+                },
+                timestamp: {
+                    gte: periodStart,
+                    lt: periodEnd,
+                },
+                action: CreditAction.SETTLE, // 必须是 SETTLE
+            },
+        });
+
+        const totalCredits = aggregation._sum?.actualCost || 0;
+        // 假设 1 Credit = $0.01 (简单换算)
+        const totalCostUSD = Number(totalCredits) * 0.01;
+
+        // 2. 落库
+        const statement = await this.prisma.billingStatement.create({
+            data: {
+                orgId,
+                periodStart,
+                periodEnd,
+                creditsConsumed: totalCredits,
+                totalCost: totalCostUSD,
+                status: 'DRAFT',
+            },
+        });
+
+        this.logger.log(`[Statement] Generated statement for Org ${orgId}: ${totalCredits} credits, $${totalCostUSD}`);
+        return statement;
+    }
+
+    private async getOrgWalletIds(orgId: string): Promise<string[]> {
+        // 1. Get Org Wallet
+        const orgWallets = await this.prisma.creditWallet.findMany({
+            where: { ownerType: 'ORGANIZATION', ownerId: orgId },
+            select: { id: true }
+        });
+
+        // 2. Get All Project IDs under Org
+        const projects = await this.prisma.project.findMany({
+            where: { organizationId: orgId },
+            select: { id: true }
+        });
+        const projectIds = projects.map(p => p.id);
+
+        // 3. Get All Project Wallets
+        const projectWallets = await this.prisma.creditWallet.findMany({
+            where: {
+                ownerType: 'PROJECT',
+                ownerId: { in: projectIds }
+            },
+            select: { id: true }
+        });
+
+        return [...orgWallets.map(w => w.id), ...projectWallets.map(w => w.id)];
+    }
+}
